Question2 Solution
Assume 3 endpoint are stored in A, B, C

Sequentially

	$http({method: 'GET', url: A, cache: 'true'}).then(function(responseA){
		$http({method: 'GET', url: B, cache: 'true'}).then(function(responseB){
			$http({method: 'GET', url: C, cache: 'true'}).then(function(responseC){
				doSomethingWithABC(responseA, responseB, responseC);
			});
		});
	});

In Parallel

	var promiseA = $http({method: 'GET', url: A, cache: 'true'});
	var promiseB = $http({method: 'GET', url: B, cache: 'true'});
	var promiseC = $http({method: 'GET', url: C, cache: 'true'});

	Promise.all([promiseA,promiseB,promiseC].then(function(response){
		doSomethingWithABC(response[0], response[1], response[2]);
	});
