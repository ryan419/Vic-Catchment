		var getResult = function(ar) {
			return ar.reduce(function(sum, item) {
				if (item instanceof Array) {
					sum = sum + getResult(item);
				} else if (typeof item === 'number') {
					sum = sum + item;
				}
				return sum;
			}, 0);
		};
